### 反向链接数量
```dataviewjs
const folder = '"pages"'
const allFiles = dv.pages(folder)

async function filenameCounter(filename){
	let count = 0
	const allmd = app.vault.getMarkdownFiles().filter(file=>{
		return file.basename != filename
	})
	
	for(let index=0; index < allmd.length; index++){
		const content = await app.vault.cachedRead(allmd[index])
		let lines = content.split("\n").forEach(line=>{
			if(line.includes(filename)){
				count++
			}
		})
	}
	return count
}


let dataArr = allFiles.map(async (file)=>{
	console.log(file)
		return [file.file.link, file.file.inlinks, await filenameCounter(file.file.name)]
	})

Promise.all(dataArr).then(values => {
	console.log(values)
	values.sort((a,b)=>b[2]-a[2])
	dv.table(["关键词", "反链", "提到次数"], values)
})
```
